CREATE TABLE friend_list(
    username1 varchar(100) CHARACTER SET utf8mb4 NOT NULL,
    username2 varchar(100) CHARACTER SET utf8mb4 NOT NULL,
    accepted int(1) unsigned NOT NULL,
    FOREIGN KEY (username1) REFERENCES user_login(username) ON DELETE CASCADE,
    FOREIGN KEY (username2) REFERENCES user_login(username) ON DELETE CASCADE,
    PRIMARY KEY (username1, username2)
) ENGINE=INNODB;

CREATE TABLE `user_login` (
 `id` int(11) unsigned NOT NULL AUTO_INCREMENT,
 `username` varchar(100) CHARACTER SET utf8mb4 NOT NULL,
 `password` varchar(100) CHARACTER SET utf8mb4 NOT NULL,
 `email` varchar(100) CHARACTER SET utf8mb4 NOT NULL,
 PRIMARY KEY (`id`),
 UNIQUE KEY `password` (`password`),
 UNIQUE KEY `username` (`username`)
) ENGINE=INNODB AUTO_INCREMENT=4 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci


CREATE TABLE `msg_table` (
 `id` int(11) unsigned NOT NULL AUTO_INCREMENT,
 `sender` varchar(100) CHARACTER SET utf8mb4 NOT NULL,
 `receiver` varchar(100) CHARACTER SET utf8mb4 NOT NULL,
 `message` varchar(10000) CHARACTER SET utf8mb4 NOT NULL,
 `date_added` date DEFAULT NULL,
 `time_added` time DEFAULT NULL,
 PRIMARY KEY (`id`)
) ENGINE=INNODB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
